                        Endpoints
                        Available request fields for generation and package discovery.
                    

                    

                        

                            

                                POST
                                /api/wx-on-demand/v1/generate
                                Generate weather content as audio or text.
                            

                            

                                
ParameterTypeDefaultDescription

                                
                                    

                                        locationsoptional
                                        string | string[]
                                        []
                                        Location IDs to resolve against configured observation, forecast, and climate entries.
                                    

                                    

                                        packagesoptional
                                        string | string[]
                                        "all"
                                        Requested package IDs in playback order. Pass "all" to include every published package.
                                    

                                    

                                        sourceoptional
                                        string | null
                                        null
                                        Restrict location resolution to a single source: eccc, nws, or twc. Auto selects the first configured match.
                                    

                                    

                                        langoptional
                                        string
                                        "en-CA"
                                        BCP-47 language tag used for text rendering and voice selection.
                                    

                                    

                                        voiceoptional
                                        string | null
                                        null
                                        Optional TTS voice override. Omit to use the default voice for the selected language.
                                    

                                    

                                        formatoptional
                                        string
                                        "raw"
                                        Audio formats return synthesized audio. Text formats return serialized package bodies instead of speech.
